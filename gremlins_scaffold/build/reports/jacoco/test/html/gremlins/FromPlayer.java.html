package gremlins;

import java.util.ArrayList;

public abstract class FromPlayer extends MovingObject {

    protected ArrayList<Gremlin> gremlins;
    private ArrayList<Slime> slimes;
    protected Gremlin collidedGremlin;
    protected Slime collidedSlime;

    public FromPlayer(int x, int y, ArrayList<StationaryObject> stonewalls, ArrayList<Brickwall> brickwalls, int speed, ArrayList<Gremlin> gremlins, ArrayList<Slime> slimes) {
        super(x, y, stonewalls, brickwalls, speed);
        this.gremlins = gremlins;
        this.slimes = slimes;
    }

    protected boolean gremlinCollision() {
        boolean collided = false;
        for (Gremlin gremlin : this.gremlins) {
            int[] gremlinSides = {gremlin.getX(), gremlin.getY(), gremlin.getX()+App.SPRITESIZE, gremlin.getY()+App.SPRITESIZE};
            if (obstacleCollision(gremlinSides, "touching")) {
                this.collidedGremlin = gremlin;
                collided = true;
                break;
            }
        }
        return collided;
    }

    protected boolean slimeCollision() {
        boolean collided = false;
        for (Slime slime : this.slimes) {
            int[] slimeSides = {slime.getX(), slime.getY(), slime.getX()+App.SPRITESIZE, slime.getY()+App.SPRITESIZE};
            if (obstacleCollision(slimeSides, "touching")) {
                this.collidedSlime = slime;
                collided = true;
                break;
            }
        }
        return collided;
    }
}
